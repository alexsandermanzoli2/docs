<ResponseField name="data" type="array">
  Array of lead objects

    </ResponseField>
    
    <ResponseField name="first_name" type="string">

    <ResponseField name="last_name" type="string">

    </ResponseField>

## Query Parameters

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of leads per page (max: 100)
</ParamField>

<ParamField query="status" type="string">
  Filter by lead status. Options: `active`, `qualified`, `unqualified`, `converted`
</ParamField>

<ParamField query="score_min" type="integer">
  Minimum lead score (0-100)
</ParamField>

<ParamField query="score_max" type="integer">
  Maximum lead score (0-100)
</ParamField>

<ParamField query="tags" type="string">
  Comma-separated list of tags to filter by
</ParamField>

<ParamField query="company_size" type="string">
  Filter by company size. Options: `1-10`, `10-50`, `50-200`, `200-1000`, `1000+`
</ParamField>

<ParamField query="industry" type="string">
  Filter by industry
</ParamField>

<ParamField query="state" type="string">
  Filter by Brazilian state code (e.g., `SP`, `RJ`, `MG`)
</ParamField>

<ParamField query="created_after" type="string">
  Filter leads created after this date (ISO 8601)
</ParamField>

<ParamField query="created_before" type="string">
  Filter leads created before this date (ISO 8601)
</ParamField>

<ParamField query="updated_after" type="string">
  Filter leads updated after this date (ISO 8601)
</ParamField>

<ParamField query="updated_before" type="string">
  Filter leads updated before this date (ISO 8601)
</ParamField>

<ParamField query="sort" type="string" default="-created_at">
  Sort field. Prefix with `-` for descending. Options: `created_at`, `updated_at`, `score`, `name`
</ParamField>

<ParamField query="fields" type="string">
  Comma-separated list of fields to include in response
</ParamField>

<ParamField query="search" type="string">
  Search leads by name, email, company, or phone
</ParamField>
