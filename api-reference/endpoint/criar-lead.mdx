---
title: 'Create Lead'
api: 'POST /v1/leads'
description: 'Create a new lead in your account'
---

## Request Body

<ParamField body="email" type="string" required>
  Lead's email address. Must be a valid email format.
</ParamField>

<ParamField body="name" type="string" required>
  Lead's full name
</ParamField>

<ParamField body="phone" type="string">
  Lead's phone number with country code (e.g., +55 11 98765-4321)
</ParamField>

<ParamField body="company" type="object">
  Company information
  
  <Expandable title="Company Object">
    <ParamField body="name" type="string" required>
      Company name
    </ParamField>
    
    <ParamField body="domain" type="string">
      Company website domain
    </ParamField>
    
    <ParamField body="cnpj" type="string">
      Company CNPJ (Brazilian tax ID)
    </ParamField>
    
    <ParamField body="industry" type="string">
      Industry classification
    </ParamField>
    
    <ParamField body="size" type="string">
      Company size: `1-10`, `10-50`, `50-200`, `200-1000`, `1000+`
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="position" type="string">
  Lead's job title
</ParamField>

<ParamField body="linkedin_url" type="string">
  Lead's LinkedIn profile URL
</ParamField>

<ParamField body="tags" type="array">
  Array of tags to associate with the lead
</ParamField>

<ParamField body="custom_fields" type="object">
  Custom field values as key-value pairs
</ParamField>

<ParamField body="validate" type="boolean" default="false">
  Whether to validate email and phone before creating
</ParamField>

<ParamField body="enrich" type="boolean" default="false">
  Whether to enrich lead data automatically
</ParamField>

## Response

<ResponseField name="data" type="object">
  The created lead object
  
  <Expandable title="Lead Object">
    <ResponseField name="id" type="string">
      Unique identifier for the lead
    </ResponseField>
    
    <ResponseField name="email" type="string">
      Lead's email address
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Lead's full name
    </ResponseField>
    
    <ResponseField name="phone" type="string">
      Lead's phone number
    </ResponseField>
    
    <ResponseField name="company" type="object">
      Company information
    </ResponseField>
    
    <ResponseField name="position" type="string">
      Lead's job title
    </ResponseField>
    
    <ResponseField name="linkedin_url" type="string">
      Lead's LinkedIn profile URL
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Lead status (default: `active`)
    </ResponseField>
    
    <ResponseField name="score" type="integer">
      Lead score (0-100)
    </ResponseField>
    
    <ResponseField name="tags" type="array">
      Array of tags
    </ResponseField>
    
    <ResponseField name="custom_fields" type="object">
      Custom field values
    </ResponseField>
    
    <ResponseField name="validation" type="object">
      Validation results if requested
      
      <Expandable title="Validation Object">
        <ResponseField name="email" type="object">
          Email validation details
        </ResponseField>
        
        <ResponseField name="phone" type="object">
          Phone validation details
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="enrichment" type="object">
      Enrichment data if requested
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
