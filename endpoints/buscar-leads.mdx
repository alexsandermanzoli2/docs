# Buscar Leads Qualificados

O endpoint de busca de leads permite encontrar prospects qualificados usando filtros avançados como localização, cargo, tamanho da empresa e setor de atuação.

<Note>
  Este endpoint está em desenvolvimento baseado na especificação da API Speedio. Para informações sobre disponibilidade, entre em contato com nossa equipe.
</Note>

## URL do Endpoint

```
GET https://api-get-leads.speedio.com.br/search_leads
```

## Parâmetros

### Query Parameters

<ParamField query="location" type="string">
  Filtro por localização (cidade, estado ou região)
  
  **Exemplos**: `"São Paulo"`, `"SP"`, `"Rio de Janeiro"`
</ParamField>

<ParamField query="job_title" type="string">
  Filtro por cargo ou posição
  
  **Exemplos**: `"CEO"`, `"Diretor"`, `"*Manager*"` (suporta wildcards)
</ParamField>

<ParamField query="company_size" type="string">
  Filtro por tamanho da empresa
  
  **Valores**: `"1-10"`, `"11-50"`, `"51-200"`, `"201-500"`, `"500+"`
</ParamField>

<ParamField query="industry" type="string">
  Filtro por setor de atuação
  
  **Exemplos**: `"Tecnologia"`, `"Saúde"`, `"Educação"`, `"Financeiro"`
</ParamField>

<ParamField query="revenue_range" type="string">
  Filtro por faixa de faturamento
  
  **Valores**: `"0-1M"`, `"1M-5M"`, `"5M-10M"`, `"10M-30M"`, `"30M+"`
</ParamField>

<ParamField query="limit" type="number" default="20">
  Limite de resultados por página (máximo 100)
</ParamField>

<ParamField query="offset" type="number" default="0">
  Número de registros para pular (paginação)
</ParamField>

## Headers Obrigatórios

<ParamField header="Authorization" type="string" required>
  Basic Authentication com suas credenciais
  
  **Formato**: `Basic base64(username:password)`
</ParamField>

<ParamField header="Content-Type" type="string">
  Tipo de conteúdo da requisição
  
  **Valor**: `application/json`
</ParamField>

```javascript Node.js
const axios = require('axios');

const username = process.env.SPEEDIO_USERNAME;
const password = process.env.SPEEDIO_PASSWORD;
const auth = Buffer.from(`${username}:${password}`).toString('base64');

async function buscarLeads(filtros) {
  try {
    const response = await axios.get(
      'https://api-get-leads.speedio.com.br/search_leads',
      {
        params: filtros,
        headers: {
          'Authorization': `Basic ${auth}`,
          'Content-Type': 'application/json'
        }
      }
    );
    
    return response.data;
  } catch (error) {
    console.error('Erro na API:', error.response?.data || error.message);
    throw error;
  }
}

// Buscar CEOs em São Paulo
const filtrosCEO = {
  job_title: 'CEO',
  location: 'São Paulo',
  company_size: '51-200',
  limit: 50
};

buscarLeads(filtrosCEO)
  .then(data => console.log('CEOs encontrados:', JSON.stringify(data, null, 2)))
  .catch(error => console.error('Erro:', error.message));
```

```python Python
import requests
import base64
import os

username = os.getenv('SPEEDIO_USERNAME')
password = os.getenv('SPEEDIO_PASSWORD')
auth = base64.b64encode(f"{username}:{password}".encode()).decode()

def buscar_leads(filtros):
    """Busca leads com filtros especificados"""
    url = "https://api-get-leads.speedio.com.br/search_leads"
    
    headers = {
        'Authorization': f'Basic {auth}',
        'Content-Type': 'application/json'
    }
    
    try:
        response = requests.get(url, params=filtros, headers=headers)
        response.raise_for_status()
        return response.json()
    except requests.exceptions.RequestException as e:
        print(f"Erro na API: {e}")
        raise

# Buscar CEOs em São Paulo
filtros_ceo = {
    'job_title': 'CEO',
    'location': 'São Paulo',
    'company_size': '51-200',
    'limit': 50
}

try:
    ceos = buscar_leads(filtros_ceo)
    print("CEOs encontrados:")
    print(json.dumps(ceos, indent=2, ensure_ascii=False))
except Exception as e:
    print(f"Erro: {e}")

# Buscar diretores em tecnologia
filtros_diretores = {
    'job_title': '*Diretor*',
    'industry': 'Tecnologia',
    'revenue_range': '10M-30M'
}

try:
    diretores = buscar_leads(filtros_diretores)
    print("Diretores encontrados:")
    print(json.dumps(diretores, indent=2, ensure_ascii=False))
except Exception as e:
    print(f"Erro: {e}")
```

## Filtros Avançados

### Wildcards em Job Title

Use wildcards para buscar variações de cargos:


### Paginação

```javascript
// Primeira página
let offset = 0;
const limit = 50;

while (true) {
  const resultado = await buscarLeads({
    job_title: 'CEO',
    limit,
    offset
  });
  
  console.log(`Página ${(offset / limit) + 1}: ${resultado.data.length} leads`);
  
  if (!resultado.pagination.has_more) break;
  
  offset = resultado.pagination.next_offset;
}
```

### 2. Análise de Mercado

```javascript
async function analisarMercado(regiao, setor) {
  const resultados = await buscarLeads({
    location: regiao,
    industry: setor,
    limit: 100
  });
  
  const analise = {
    total: resultados.pagination.total,
    empresas_unicas: new Set(resultados.data.map(l => l.company.cnpj)).size,
    cargos_frequentes: {},
    tamanhos_empresa: {},
    score_medio: 0
  };
  
  resultados.data.forEach(lead => {
    // Análise de cargos
    analise.cargos_frequentes[lead.job_title] = 
      (analise.cargos_frequentes[lead.job_title] || 0) + 1;
    
    // Análise de tamanhos
    analise.tamanhos_empresa[lead.company.size] = 
      (analise.tamanhos_empresa[lead.company.size] || 0) + 1;
    
    // Score médio
    analise.score_medio += lead.score;
  });
  
  analise.score_medio = analise.score_medio / resultados.data.length;
  
  return analise;
}

// Analisar mercado de tecnologia em SP
const analise = await analisarMercado('São Paulo', 'Tecnologia');
console.log('Análise do mercado:', analise);
```

### 3. Lead Scoring Personalizado

```javascript
function calcularScorePersonalizado(lead) {
  let score = lead.score; // Score base da API
  
  // Bônus por verificação
  if (lead.verified.email) score += 10;
  if (lead.verified.phone) score += 10;
  if (lead.verified.linkedin) score += 5;
  
  // Bônus por cargo
  if (lead.seniority === 'c_level') score += 20;
  else if (lead.seniority === 'director') score += 15;
  else if (lead.seniority === 'manager') score += 10;
  
  // Bônus por tamanho da empresa
  const tamanho = lead.company.size;
  if (tamanho === '201-500' || tamanho === '500+') score += 15;
  else if (tamanho === '51-200') score += 10;
  
  // Bônus por atividade recente
  const ultimaAtividade = new Date(lead.last_activity);
  const agora = new Date();
  const diasInativo = (agora - ultimaAtividade) / (1000 * 60 * 60 * 24);
  
  if (diasInativo < 30) score += 10;
  else if (diasInativo < 90) score += 5;
  
  return Math.min(score, 100); // Máximo 100
}

// Aplicar scoring personalizado
const leads = await buscarLeads({ job_title: 'CEO', limit: 50 });
const leadsComScore = leads.data.map(lead => ({
  ...lead,
  score_personalizado: calcularScorePersonalizado(lead)
}));

// Ordenar por score personalizado
leadsComScore.sort((a, b) => b.score_personalizado - a.score_personalizado);
```

## Limites e Considerações

### Rate Limits
- **50 requisições por minuto** para busca de leads
- **500 requisições por hora** por conta

### Performance
- **Tempo de resposta**: 300-800ms por requisição
- **Máximo**: 100 leads por página
- **Recomendação**: Use filtros específicos para melhor performance

### Custos
- **1 crédito** por página de resultados (até 100 leads)
- Filtros não consomem créditos adicionais
